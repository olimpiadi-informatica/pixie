1//! The database for pixie-server.
2//!
3//! Structure of the storage directory:
4//! - `config.yaml`: configuration file for pixie-server
5//! - `registered.json`: json file containing all information about registered units.
6//! - `admin/`: directory containing the static files for the admin web interface.
7//! - `chunks/`: directory containing the image's chunks.
8//! - `images/`: directory containing the image's info.
9//! - `tftpboot/`: directory containing the necessary files for network boot.

40/// Atomically write `data` at the specified `path`.
41///
42/// On crash `path` is guaranteed to be in a consistent state, but a temporary file might be left
43/// behind.
44fn atomic_write(path: &Path, data: &[u8]) -> Result<()> {
45    static CNT: AtomicU64 = AtomicU64::new(0);
46
47    let (tmp_path, mut file) = loop {
48        let tmp_path =
49            path.with_file_name(format!("{:x}.tmp", CNT.fetch_add(1, Ordering::Relaxed)));
50        match File::options().write(true).create_new(true).open(&tmp_path) {
51            Ok(file) => break (tmp_path, file),
52            Err(e) if e.kind() == ErrorKind::AlreadyExists => continue,
53            Err(e) => Err(e)?,
54        }
55    };
56    file.write_all(data)
57        .with_context(|| format!("write to file: {}", tmp_path.display()))?;
58    std::mem::drop(file);
59    std::fs::rename(&tmp_path, path)?;
60    Ok(())
61}

63/// Builds a map from ip address to hostname parsing the hostfile at `path`.
64fn build_hostmap(path: Option<&Path>) -> Result<HashMap<Ipv4Addr, String>> {
65    let mut hostmap = HashMap::new();
66    if let Some(path) = path {
67        let hosts =
68            hostfile::parse_file(path).map_err(|e| anyhow!("Error parsing host file: {e}"))?;
69        for mut host in hosts {
70            if let IpAddr::V4(ip) = host.ip {
71                let old = hostmap.insert(ip, std::mem::take(&mut host.names[0]));
72                if old.is_some() {
73                    log::warn!("Duplicated hostname for {ip}");
74                }
75            } else {
76                log::warn!(
77                    "ignoring non-IPv4 address {} for host {}",
78                    host.ip,
79                    host.names[0]
80                );
81            }
82        }
83    }
84    Ok(hostmap)
85}

87/// See [the module-level documentation][self].
88pub struct State {
89    /// The storage_dir received by command line arguments.
90    pub storage_dir: PathBuf,
91    /// A directory stored in ram for dynamically generated files, will be deleted on Drop.
92    pub run_dir: PathBuf,
93    /// The config parsed from the config file.
94    pub config: Config,
95    /// The hostmap built from the hostmap file.
96    hostmap: watch::Sender<HashMap<Ipv4Addr, String>>,
97
98    units: watch::Sender<Vec<Unit>>,
99    registration_hint: Mutex<Option<RegistrationInfo>>,
100    images_stats: watch::Sender<ImagesStats>,
101    chunks_stats: Mutex<ChunksStats>,
102}
103
104impl State {
105    /// Loads the [`State`] from the given path.
106    pub fn load(storage_dir: PathBuf) -> Result<Self> {
107        let config: Config = {
108            let path = storage_dir.join(CONFIG_YAML);
109            let file = File::open(&path)
110                .with_context(|| format!("open config file: {}", path.display()))?;
111            serde_yaml::from_reader(file)
112                .with_context(|| format!("deserialize config from {}", path.display()))?
113        };
114
115        let hostmap = build_hostmap(config.hosts.hostsfile.as_deref())?;
116
117        let units_path = storage_dir.join(REGISTERED_JSON);
118        let units: Vec<Unit> = if units_path.exists() {
119            let file = File::open(&units_path)
120                .with_context(|| format!("open units file: {}", units_path.display()))?;
121            serde_json::from_reader(&file)
122                .with_context(|| format!("deserialize units from {}", units_path.display()))?
123        } else {
124            Vec::new()
125        };
126        for unit in &units {
127            ensure!(
128                config.groups.get_by_second(&unit.group).is_some(),
129                "unit group {} not found in config",
130                unit.group,
131            );
132            ensure!(
133                config.images.contains(&unit.image),
134                "unit image {} not found in config",
135                unit.image,
136            );
137        }
138        let units = watch::Sender::new(units);
139
140        let mut units_rx = units.subscribe();
141        tokio::spawn(async move {
142            while units_rx.changed().await.is_ok() {
143                let units = units_rx.borrow_and_update().clone();
144                let json = serde_json::to_vec(&units).expect("serialize units");
145                // TODO(virv): handle error
146                atomic_write(&units_path, &json).expect("write units file");
147            }
148        });
149
150        let chunks_dir = storage_dir.join(CHUNKS_DIR);
151        let mut chunks_stats: ChunksStats = std::fs::read_dir(&chunks_dir)
152            .with_context(|| format!("open chunks dir: {}", chunks_dir.display()))?
153            .map(|file| {
154                let file = file?;
155                let metadata = file.metadata()?;
156                let csize = metadata.len();
157
158                let name = file
159                    .file_name()
160                    .to_str()
161                    .and_then(|s| hex::decode(s).ok())
162                    .and_then(|s| ChunkHash::try_from(&s[..]).ok())
163                    .with_context(|| format!("invalid chunk name: {:?}", file.file_name()))?;
164
165                Ok((name, ChunkStats { csize, ref_cnt: 0 }))
166            })
167            .collect::<Result<_>>()?;
168
169        let images_dir = storage_dir.join(IMAGES_DIR);
170        let images = std::fs::read_dir(&images_dir)
171            .with_context(|| format!("open images dir: {}", images_dir.display()))?
172            .map(|image_entry| {
173                let image_entry = image_entry?;
174                let path = image_entry.path();
175                let image_name = image_entry
176                    .file_name()
177                    .into_string()
178                    .map_err(|_| anyhow!("invalid image name {:?}", image_entry.file_name()))?;
179                let content = std::fs::read(&path)
180                    .with_context(|| format!("read image file: {}", path.display()))?;
181                let image = postcard::from_bytes::<Image>(&content)
182                    .with_context(|| format!("deserialize image from {}", path.display()))?;
183                for chunk in &image.disk {
184                    let chunk_stats = chunks_stats
185                        .get_mut(&chunk.hash)
186                        .with_context(|| format!("chunk {} not found", hex::encode(chunk.hash)))?;
187                    chunk_stats.ref_cnt += 1;
188                }
189                Ok((image_name, (image.size(), image.csize())))
190            })
191            .collect::<Result<_>>()?;
192
193        let reclaimable = chunks_stats
194            .values()
195            .filter(|stat| stat.ref_cnt == 0)
196            .map(|stat| stat.csize)
197            .sum();
198        let total_csize = chunks_stats.values().map(|stat| stat.csize).sum();
199
200        let images_stats = ImagesStats {
201            total_csize,
202            reclaimable,
203            images,
204        };
205
206        let run_dir = PathBuf::from(format!("/run/pixie-{}", std::process::id()));
207        std::fs::create_dir(&run_dir)?;
208
209        Ok(Self {
210            storage_dir,
211            run_dir,
212            config,
213            hostmap: watch::Sender::new(hostmap),
214            units,
215            registration_hint: Mutex::new(None),
216            images_stats: watch::Sender::new(images_stats),
217            chunks_stats: Mutex::new(chunks_stats),
218        })
219    }
220
221    pub fn reload(&self) -> Result<()> {
222        let hostmap = build_hostmap(self.config.hosts.hostsfile.as_deref())?;
223        self.hostmap.send_replace(hostmap);
224        Ok(())
225    }
226
227    pub fn subscribe_hostmap(&self) -> watch::Receiver<HashMap<Ipv4Addr, String>> {
228        self.hostmap.subscribe()
229    }
230}

232impl Drop for State {
233    fn drop(&mut self) {
234        // If we fail to remove the directory it's not an issue as it will be deleted on shutdown
235        // and doesn't take much space.
236        let _ = std::fs::remove_dir_all(&self.run_dir);
237    }
