24async fn broadcast_chunks(
25    state: &State,
26    socket: &UdpSocket,
27    ip: Ipv4Addr,
28    mut rx: Receiver<ChunkHash>,
29) -> Result<()> {
30    let mut queue = BTreeSet::<ChunkHash>::new();
31    let mut write_buf = [0; PACKET_LEN];
32    let mut wait_for = Instant::now();
33    let mut index = [0; 32];
34
35    loop {
36        let get_index = async {
37            while let Ok(hash) = rx.try_recv() {
38                queue.insert(hash);
39            }
40
41            let hash = queue
42                .range((Bound::Excluded(index), Bound::Unbounded))
43                .next()
44                .or_else(|| queue.iter().next())
45                .copied();
46
47            match hash {
48                Some(hash) => {
49                    queue.remove(&hash);
50                    Some(hash)
51                }
52                None => rx.recv().await,
53            }
54        };
55
56        tokio::select! {
57            hash = get_index => {
58                let Some(hash) = hash else {
59                    break;
60                };
61                index = hash;
62            }
63            _ = state.cancel_token.cancelled() => break,
64        };
65
66        let Some(data) = state
67            .get_chunk_cdata(index)
68            .with_context(|| format!("get chunk {}", hex::encode(index)))?
69        else {
70            log::warn!("Chunk {} not found", hex::encode(index));
71            continue;
72        };
73
74        let num_packets = data.len().div_ceil(BODY_LEN);
75        write_buf[..32].clone_from_slice(&index);
76
77        let mut xor = [[0; BODY_LEN]; 32];
78
79        let hosts_cfg = &state.config.hosts;
80        let chunks_addr = SocketAddrV4::new(ip, CHUNKS_PORT);
81
82        for index in 0..num_packets {
83            write_buf[32..34].clone_from_slice(&(index as u16).to_le_bytes());
84            let start = index * BODY_LEN;
85            let len = BODY_LEN.min(data.len() - start);
86            let body = &data[start..start + len];
87            let group = index & 31;
88            body.iter()
89                .zip(xor[group].iter_mut())
90                .for_each(|(a, b)| *b ^= a);
91            write_buf[34..34 + len].clone_from_slice(body);
92
93            wait_for = wait_for.max(Instant::now());
94            time::sleep_until(wait_for).await;
95
96            let sent_len = socket.send_to(&write_buf[..34 + len], chunks_addr).await?;
97            ensure!(sent_len == 34 + len, "Could not send packet");
98            wait_for += 8 * (sent_len as u32) * Duration::from_secs(1) / hosts_cfg.broadcast_speed;
99        }
100
101        for (index, body) in xor.iter().enumerate().take(num_packets) {
102            write_buf[32..34].clone_from_slice(&(index as u16).wrapping_sub(32).to_le_bytes());
103            let len = BODY_LEN;
104            write_buf[34..34 + len].clone_from_slice(body);
105
106            wait_for = wait_for.max(Instant::now());
107            time::sleep_until(wait_for).await;
108            let sent_len = socket.send_to(&write_buf[..34 + len], chunks_addr).await?;
109            ensure!(sent_len == 34 + len, "Could not send packet");
110            wait_for += 8 * (sent_len as u32) * Duration::from_secs(1) / hosts_cfg.broadcast_speed;
111        }
112    }
113
114    Ok(())
115}

117fn compute_hint(state: &State) -> Result<RegistrationInfo> {
118    let Some(mut last) = state.get_registration_hint() else {
119        return Ok(RegistrationInfo {
120            group: state
121                .config
122                .groups
123                .iter()
124                .next()
125                .context("there should be at least one group")?
126                .0
127                .clone(),
128            row: 1,
129            col: 1,
130            image: state
131                .config
132                .images
133                .first()
134                .context("there should be at least one image")?
135                .clone(),
136        });
137    };
138
139    let units = state.get_units(UnitSelector::Group(
140        *state.config.groups.get_by_first(&last.group).unwrap(),
141    ));
142    let positions = units
143        .into_iter()
144        .map(|unit| (unit.row, unit.col))
145        .collect::<Vec<_>>();
146
147    if last.row == 0 {
148        if let Some(&(r, c)) = positions.iter().max() {
149            last.row = r;
150            last.col = c;
151        }
152    }
153
154    let (mrow, mcol) = positions
155        .iter()
156        .fold((0, 0), |(r1, c1), &(r2, c2)| (r1.max(r2), c1.max(c2)));
157
158    let (row, col) = match mrow {
159        0 => (1, 1),
160        1 => (1, mcol + 1),
161        _ => (last.row + last.col / mcol, last.col % mcol + 1),
162    };
163
164    Ok(RegistrationInfo {
165        group: last.group.clone(),
166        row,
167        col,
168        image: last.image.clone(),
169    })
170}

172async fn broadcast_hint(state: &State, socket: &UdpSocket, ip: Ipv4Addr) -> Result<()> {
173    loop {
174        tokio::select! {
175            _ = time::sleep(Duration::from_secs(1)) => {}
176            _ = state.cancel_token.cancelled() => break,
177        }
178        let hint = HintPacket {
179            station: compute_hint(state)?,
180            images: state.config.images.clone(),
181            groups: state.config.groups.clone(),
182        };
183        let data = postcard::to_allocvec(&hint)?;
184        let hint_addr = SocketAddrV4::new(ip, HINT_PORT);
185        socket.send_to(&data, hint_addr).await?;
186    }
187    Ok(())
188}

190async fn handle_requests(state: &State, socket: &UdpSocket, tx: Sender<[u8; 32]>) -> Result<()> {
191    let mut buf = [0; PACKET_LEN];
192    loop {
193        let (len, addr) = tokio::select! {
194            x = socket.recv_from(&mut buf) => x?,
195            _ = state.cancel_token.cancelled() => break,
196        };
197        let req: postcard::Result<UdpRequest> = postcard::from_bytes(&buf[..len]);
198        match req {
199            Ok(UdpRequest::Discover) => {
200                socket.send_to(&[], addr).await?;
201            }
202            Ok(UdpRequest::ActionProgress(frac, tot)) => {
203                let IpAddr::V4(peer_ip) = addr.ip() else {
204                    bail!("IPv6 is not supported")
205                };
206                match find_mac(peer_ip) {
207                    Ok(peer_mac) => {
208                        state.set_unit_progress(UnitSelector::MacAddr(peer_mac), Some((frac, tot)));
209                    }
210                    Err(err) => {
211                        log::error!("Error handling udp packet: {err}");
212                    }
213                };
214            }
215            Ok(UdpRequest::RequestChunks(chunks)) => {
216                for hash in chunks {
217                    tx.send(hash).await?;
218                }
219            }
220            Err(e) => {
221                log::warn!("Invalid request from {addr}: {e}");
222            }
223        }
224    }
225    Ok(())
226}

228pub async fn main(state: Arc<State>) -> Result<()> {
229    let (_, network) = find_network(state.config.hosts.listen_on)?;
230
231    let (tx, rx) = mpsc::channel(128);
232    let socket = UdpSocket::bind((Ipv4Addr::UNSPECIFIED, ACTION_PORT)).await?;
233    log::info!("Listening on {}", socket.local_addr()?);
234    socket.set_broadcast(true)?;
235
236    tokio::try_join!(
237        broadcast_chunks(&state, &socket, network.broadcast(), rx),
238        broadcast_hint(&state, &socket, network.broadcast()),
239        handle_requests(&state, &socket, tx),
240    )?;
241
242    Ok(())
243}
