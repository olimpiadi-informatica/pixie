20async fn handle_request(state: &State, req: TcpRequest, peer_mac: MacAddr6) -> Result<Vec<u8>> {
21    Ok(match req {
22        TcpRequest::GetChunkSize(hash) => {
23            let path = state.storage_dir.join("chunks").join(hex::encode(hash));
24            let csize = match fs::metadata(path).await {
25                Ok(meta) => Some(meta.len()),
26                Err(e) if e.kind() == ErrorKind::NotFound => None,
27                Err(e) => Err(e)?,
28            };
29            postcard::to_allocvec(&csize)?
30        }
31        TcpRequest::GetImage(name) => {
32            let path = state.storage_dir.join("images").join(name);
33            fs::read(path).await?
34        }
35        TcpRequest::Register(station) => {
36            if !state.config.images.contains(&station.image) {
37                return Ok(format!("Unknown image: {}", station.image).into_bytes());
38            }
39            let Some(&group) = state.config.groups.get_by_first(&station.group) else {
40                return Ok(format!("Unknown group: {}", station.group).into_bytes());
41            };
42
43            let mut guard = state
44                .last
45                .lock()
46                .map_err(|_| anyhow!("last mutex is poisoned"))?;
47            *guard = station.clone();
48
49            state.units.send_modify(|units| {
50                let unit = units.iter_mut().position(|unit| unit.mac == peer_mac);
51                match unit {
52                    Some(unit) => {
53                        units[unit].group = group;
54                        units[unit].row = station.row;
55                        units[unit].col = station.col;
56                        units[unit].image = station.image;
57                    }
58                    None => {
59                        let unit = Unit {
60                            mac: peer_mac,
61                            group,
62                            row: station.row,
63                            col: station.col,
64                            curr_action: None,
65                            curr_progress: None,
66                            next_action: ActionKind::Wait,
67                            image: station.image,
68                            last_ping_timestamp: 0,
69                            last_ping_msg: Vec::new(),
70                        };
71                        units.push(unit);
72                    }
73                }
74            });
75
76            Vec::new()
77        }
78        TcpRequest::UploadChunk(hash, data) => {
79            state.add_chunk(hash, &data)?;
80            Vec::new()
81        }
82        TcpRequest::UploadImage(name, image) => {
83            if !state.config.images.contains(&name) {
84                return Ok(format!("Unknown image: {}", name).into_bytes());
85            }
86            state.add_image(name, &image)?;
87
88            Vec::new()
89        }
90        TcpRequest::GetAction => {
91            let mut action = Action::Wait;
92
93            state.units.send_if_modified(|units| {
94                let mut unit = units.iter_mut().find(|unit| unit.mac == peer_mac);
95
96                let modified;
97                let action_kind;
98
99                match unit {
100                    Some(Unit {
101                        curr_action: Some(action),
102                        ..
103                    }) => {
104                        action_kind = *action;
105                        modified = false;
106                    }
107                    Some(ref mut unit) => {
108                        match unit.next_action {
109                            ActionKind::Push | ActionKind::Pull | ActionKind::Register => {
110                                unit.curr_action = Some(unit.next_action);
111                                unit.next_action = ActionKind::Wait;
112                                modified = true;
113                            }
114                            ActionKind::Reboot | ActionKind::Wait => {
115                                modified = false;
116                            }
117                        }
118                        action_kind = unit.next_action;
119                    }
120                    None => {
121                        action_kind = ActionKind::Register;
122                        modified = false;
123                    }
124                }
125
126                action = match action_kind {
127                    ActionKind::Reboot => Action::Reboot,
128                    ActionKind::Register => Action::Register,
129                    ActionKind::Push => Action::Push {
130                        image: unit.unwrap().image.clone(),
131                    },
132                    ActionKind::Pull => Action::Pull {
133                        image: unit.unwrap().image.clone(),
134                    },
135                    ActionKind::Wait => Action::Wait,
136                };
137
138                modified
139            });
140
141            postcard::to_allocvec(&action)?
142        }
143        TcpRequest::ActionComplete => {
144            state.units.send_if_modified(|units| {
145                let Some(unit) = units.iter_mut().find(|unit| unit.mac == peer_mac) else {
146                    log::warn!("Got ActionComplete from unknown unit");
147                    return false;
148                };
149
150                unit.curr_action = None;
151                unit.curr_progress = None;
152                true
153            });
154
155            Vec::new()
156        }
157    })
158}

160async fn handle_connection(
161    state: Arc<State>,
162    mut stream: TcpStream,
163    peer_addr: SocketAddr,
164) -> Result<()> {
165    let IpAddr::V4(peer_ip) = peer_addr.ip() else {
166        bail!("IPv6 is not supported")
167    };
168    let peer_mac = match find_mac(peer_ip) {
169        Ok(peer_mac) => peer_mac,
170        Err(err) => {
171            log::error!("Error handling tcp connection: {}", err);
172            return Ok(());
173        }
174    };
175
176    loop {
177        let len = match stream.read_u64_le().await {
178            Ok(len) => len as usize,
179            Err(e) if e.kind() == ErrorKind::ConnectionReset => return Ok(()),
180            Err(e) => Err(e)?,
181        };
182        let mut buf = vec![0; len];
183        stream.read_exact(&mut buf).await?;
184        let req = postcard::from_bytes(&buf)?;
185        let resp = handle_request(&state, req, peer_mac).await?;
186        stream.write_u64_le(resp.len() as u64).await?;
187        stream.write_all(&resp).await?;
188    }
189}

191pub async fn main(state: Arc<State>) -> Result<()> {
192    let listener = TcpListener::bind((Ipv4Addr::UNSPECIFIED, ACTION_PORT)).await?;
193    log::info!("Listening on {}", listener.local_addr()?);
194    loop {
195        let (stream, addr) = listener.accept().await?;
196        let state = state.clone();
197        tokio::spawn(async move {
198            if let Err(e) = handle_connection(state, stream, addr).await {
199                log::error!("Error handling tcp connection: {}", e);
200            }
201        });
202    }
203}
