18async fn handle_request(state: &State, req: TcpRequest, peer_mac: MacAddr6) -> Result<Vec<u8>> {
19    Ok(match req {
20        TcpRequest::GetChunkCSize(hash) => {
21            let csize = state.get_chunk_csize(hash);
22            postcard::to_allocvec(&csize)?
23        }
24        TcpRequest::GetImage(name) => state.get_image_serialized(&name)?.unwrap(),
25        TcpRequest::Register(station) => {
26            state.set_last(station.clone());
27            state.register_unit(peer_mac, station)?;
28            Vec::new()
29        }
30        TcpRequest::UploadChunk(data) => {
31            state.add_chunk(&data)?;
32            Vec::new()
33        }
34        TcpRequest::UploadImage(name, image) => {
35            state.add_image(name, &image)?;
36            Vec::new()
37        }
38        TcpRequest::GetAction => {
39            let action = state.get_unit_action(peer_mac);
40            postcard::to_allocvec(&action)?
41        }
42        TcpRequest::ActionComplete => {
43            state.unit_complete_action(UnitSelector::MacAddr(peer_mac));
44            Vec::new()
45        }
46    })
47}

49async fn handle_connection(
50    state: Arc<State>,
51    mut stream: TcpStream,
52    peer_addr: SocketAddr,
53) -> Result<()> {
54    let IpAddr::V4(peer_ip) = peer_addr.ip() else {
55        bail!("IPv6 is not supported")
56    };
57    let peer_mac = match find_mac(peer_ip) {
58        Ok(peer_mac) => peer_mac,
59        Err(err) => {
60            log::error!("Error handling tcp connection: {}", err);
61            return Ok(());
62        }
63    };
64
65    loop {
66        let len = match stream.read_u64_le().await {
67            Ok(len) => len as usize,
68            Err(e) if e.kind() == ErrorKind::ConnectionReset => return Ok(()),
69            Err(e) => Err(e)?,
70        };
71        let mut buf = vec![0; len];
72        stream.read_exact(&mut buf).await?;
73        let req = postcard::from_bytes(&buf)?;
74        let resp = handle_request(&state, req, peer_mac).await?;
75        stream.write_u64_le(resp.len() as u64).await?;
76        stream.write_all(&resp).await?;
77    }
78}

80pub async fn main(state: Arc<State>) -> Result<()> {
81    let listener = TcpListener::bind((Ipv4Addr::UNSPECIFIED, ACTION_PORT)).await?;
82    log::info!("Listening on {}", listener.local_addr()?);
83    loop {
84        let (stream, addr) = listener.accept().await?;
85        let state = state.clone();
86        tokio::spawn(async move {
87            if let Err(e) = handle_connection(state, stream, addr).await {
88                log::error!("Error handling tcp connection: {}", e);
89            }
90        });
91    }
92}
