16    pub start: Offset,
17    pub size: usize,

20async fn save_image(stream: &TcpStream, image: Image) -> Result<()> {
21    let req = TcpRequest::UploadImage(image);
22    let buf = postcard::to_allocvec(&req)?;
23    stream.send_u64_le(buf.len() as u64).await?;
24    stream.send(&buf).await?;
25    let len = stream.recv_u64_le().await?;
26    assert_eq!(len, 0);
27    Ok(())
28}

30enum State {
31    ReadingPartitions,
32    PushingChunks {
33        cur: usize,
34        total: usize,
35        tsize: usize,
36        tcsize: usize,
37    },
38}

40pub async fn store(os: UefiOS, server_address: SocketAddrV4) -> Result<()> {
41    let stats = Rc::new(RefCell::new(State::ReadingPartitions));
42    let stats2 = stats.clone();
43    os.set_ui_drawer(move |os| match &*stats2.borrow() {
44        State::ReadingPartitions => {
45            os.write_with_color("Reading partitions...", Color::White, Color::Black)
46        }
47        State::PushingChunks {
48            cur,
49            total,
50            tsize,
51            tcsize,
52        } => {
53            os.write_with_color(
54                &format!("Pushed {cur} out of {total} chunks\n"),
55                Color::White,
56                Color::Black,
57            );
58            os.write_with_color(
59                &format!("total size {tsize}, compressed {tcsize}\n"),
60                Color::White,
61                Color::Black,
62            );
63        }
64    });
65
66    let bo = os.boot_options();
67    let boid = bo.reboot_target().expect("Could not find reboot target");
68    let bo_command = bo.get(boid);
69
70    let mut disk = os.open_first_disk();
71    let chunks = parse_disk::parse_disk(&mut disk).await?;
72
73    // Split up chunks.
74    let mut final_chunks = Vec::<ChunkInfo>::new();
75    for ChunkInfo { mut start, size } in chunks {
76        let end = start + size;
77
78        if let Some(last) = final_chunks.last() {
79            assert!(last.start + last.size <= start);
80            if last.start + last.size == start {
81                start = last.start;
82                final_chunks.pop();
83            }
84        }
85
86        while start < end {
87            let split = (start + 1).next_multiple_of(MAX_CHUNK_SIZE).min(end);
88            final_chunks.push(ChunkInfo {
89                start,
90                size: split - start,
91            });
92            start = split;
93        }
94    }
95
96    let udp = os.udp_bind(None).await?;
97    let stream_get_csize = os.connect(server_address).await?;
98    let stream_upload_chunk = os.connect(server_address).await?;
99
100    let total = final_chunks.len();
101
102    let mut total_size = 0;
103    let mut total_csize = 0;
104
105    let total_mem = os.get_total_mem();
106    let channel_size =
107        (total_mem.saturating_sub(MIN_MEMORY) as usize / (4 * MAX_CHUNK_SIZE)).max(32);
108    log::debug!("Total memory: {total_mem}. Channel size: {channel_size}");
109
110    let (tx1, mut rx1) = mpsc::channel(channel_size);
111    let (tx2, mut rx2) = mpsc::channel(channel_size);
112    let (tx3, mut rx3) = mpsc::channel(channel_size);
113    let (tx4, mut rx4) = mpsc::channel(channel_size);
114
115    let task1 = async {
116        let mut tx1 = tx1;
117        for chunk_info in final_chunks {
118            let mut data = vec![0; chunk_info.size];
119            disk.read(chunk_info.start as u64, &mut data).await?;
120            let cdata = compress(&data);
121            let hash = blake3::hash(&data).into();
122            let chunk = Chunk {
123                hash,
124                start: chunk_info.start,
125                size: chunk_info.size,
126                csize: cdata.len(),
127            };
128            tx1.send((chunk, cdata)).await;
129        }
130        Ok::<_, Error>(())
131    };
132
133    let task2 = async {
134        let mut tx2 = tx2;
135        while let Some((chunk, cdata)) = rx1.recv().await {
136            let req = TcpRequest::HasChunk(chunk.hash);
137            let buf = postcard::to_allocvec(&req)?;
138            stream_get_csize.send_u64_le(buf.len() as u64).await?;
139            stream_get_csize.send(&buf).await?;
140            tx2.send((chunk, cdata)).await;
141        }
142        Ok(())
143    };
144
145    let task3 = async {
146        let mut tx3 = tx3;
147        while let Some((chunk, cdata)) = rx2.recv().await {
148            let len = stream_get_csize.recv_u64_le().await?;
149            let mut buf = vec![0; len as usize];
150            stream_get_csize.recv(&mut buf).await?;
151            let has_chunk: bool = postcard::from_bytes(&buf)?;
152            tx3.send((chunk, cdata, has_chunk)).await;
153        }
154        Ok(())
155    };
156
157    let task4 = async {
158        let mut tx4 = tx4;
159        while let Some((chunk, cdata, has_chunk)) = rx3.recv().await {
160            if !has_chunk {
161                let req = TcpRequest::UploadChunk(cdata);
162                let buf = postcard::to_allocvec(&req)?;
163                stream_upload_chunk.send_u64_le(buf.len() as u64).await?;
164                stream_upload_chunk.send(&buf).await?;
165            }
166            tx4.send((chunk, has_chunk)).await;
167        }
168        Ok(())
169    };
170
171    let task5 = async {
172        stats.replace(State::PushingChunks {
173            cur: 0,
174            total,
175            tsize: 0,
176            tcsize: 0,
177        });
178
179        let mut chunks = Vec::new();
180        while let Some((chunk, has_chunk)) = rx4.recv().await {
181            if !has_chunk {
182                let len = stream_upload_chunk.recv_u64_le().await?;
183                assert_eq!(len, 0);
184            }
185            chunks.push(chunk);
186
187            total_size += chunk.size;
188            total_csize += chunk.csize;
189
190            stats.replace(State::PushingChunks {
191                cur: chunks.len(),
192                total,
193                tsize: total_size,
194                tcsize: total_csize,
195            });
196            udp.send(
197                server_address,
198                &postcard::to_allocvec(&UdpRequest::ActionProgress(chunks.len(), total))?,
199            )
200            .await?;
201        }
202        Ok(chunks)
203    };
204
205    let ((), (), (), (), chunk_hashes) = futures::try_join!(task1, task2, task3, task4, task5)?;
206
207    save_image(
208        &stream_upload_chunk,
209        Image {
210            boot_option_id: boid,
211            boot_entry: bo_command,
212            disk: chunk_hashes,
213        },
214    )
215    .await?;
216
217    stream_get_csize.close_send().await;
218    stream_get_csize.force_close().await;
219
220    stream_upload_chunk.close_send().await;
221    // TODO(virv): this could be better
222    stream_upload_chunk.force_close().await;
223
224    log::info!("image saved. Total size {total_size}, total csize {total_csize}");
225
226    Ok(())
227}
