21async fn save_image(stream: &TcpStream, image: Image) -> Result<()> {
22    let req = TcpRequest::UploadImage(image);
23    let buf = postcard::to_allocvec(&req)?;
24    stream.send_u64_le(buf.len() as u64).await?;
25    stream.send(&buf).await?;
26    let len = stream.recv_u64_le().await?;
27    assert_eq!(len, 0);
28    Ok(())
29}

36        tsize: usize,
37        tcsize: usize,
38    },

41pub async fn store(os: UefiOS, server_address: SocketAddrV4) -> Result<()> {
42    let stats = Rc::new(RefCell::new(State::ReadingPartitions));
43    let stats2 = stats.clone();
44    os.set_ui_drawer(move |os| match &*stats2.borrow() {
45        State::ReadingPartitions => {
46            os.write_with_color("Reading partitions...", Color::White, Color::Black)
47        }
48        State::PushingChunks {
49            cur,
50            total,
51            tsize,
52            tcsize,
53        } => {
54            os.write_with_color(
55                &format!("Pushed {cur} out of {total} chunks\n"),
56                Color::White,
57                Color::Black,
58            );
59            os.write_with_color(
60                &format!("total size {tsize}, compressed {tcsize}\n"),
61                Color::White,
62                Color::Black,
63            );
64        }
65    });
66
67    let bo = os.boot_options();
68    let boid = bo.reboot_target().expect("Could not find reboot target");
69    let bo_command = bo.get(boid);
70
71    let mut disk = os.open_first_disk();
72    let chunks = parse_disk::parse_disk(&mut disk).await?;
73    info!(
74        "Total size of chunks: {}",
75        BytesFmt(chunks.iter().map(|x| x.size as u64).sum::<u64>())
76    );
77
78    let udp = os.udp_bind(None).await?;
79    let stream_get_csize = os.connect(server_address).await?;
80    let stream_upload_chunk = os.connect(server_address).await?;
81
82    let total = chunks.len();
83
84    let mut total_size = 0;
85    let mut total_csize = 0;
86
87    let total_mem = os.get_total_mem();
88    let channel_size =
89        (total_mem.saturating_sub(MIN_MEMORY) as usize / (4 * MAX_CHUNK_SIZE)).max(32);
90    log::debug!("Total memory: {total_mem}. Channel size: {channel_size}");
91
92    let (tx1, mut rx1) = mpsc::channel(channel_size);
93    let (tx2, mut rx2) = mpsc::channel(channel_size);
94    let (tx3, mut rx3) = mpsc::channel(channel_size);
95    let (tx4, mut rx4) = mpsc::channel(channel_size);
96
97    let task1 = async {
98        let mut tx1 = tx1;
99        for chunk_info in chunks {
100            let mut data = vec![0; chunk_info.size];
101            disk.read(chunk_info.start as u64, &mut data).await?;
102            let cdata = compress(&data);
103            let hash = blake3::hash(&data).into();
104            let chunk = Chunk {
105                hash,
106                start: chunk_info.start,
107                size: chunk_info.size,
108                csize: cdata.len(),
109            };
110            tx1.send((chunk, cdata)).await;
111        }
112        Ok::<_, Error>(())
113    };
114
115    let task2 = async {
116        let mut tx2 = tx2;
117        while let Some((chunk, cdata)) = rx1.recv().await {
118            let req = TcpRequest::HasChunk(chunk.hash);
119            let buf = postcard::to_allocvec(&req)?;
120            stream_get_csize.send_u64_le(buf.len() as u64).await?;
121            stream_get_csize.send(&buf).await?;
122            tx2.send((chunk, cdata)).await;
123        }
124        Ok(())
125    };
126
127    let task3 = async {
128        let mut tx3 = tx3;
129        while let Some((chunk, cdata)) = rx2.recv().await {
130            let len = stream_get_csize.recv_u64_le().await?;
131            let mut buf = vec![0; len as usize];
132            stream_get_csize.recv(&mut buf).await?;
133            let has_chunk: bool = postcard::from_bytes(&buf)?;
134            tx3.send((chunk, cdata, has_chunk)).await;
135        }
136        Ok(())
137    };
138
139    let task4 = async {
140        let mut tx4 = tx4;
141        while let Some((chunk, cdata, has_chunk)) = rx3.recv().await {
142            if !has_chunk {
143                let req = TcpRequest::UploadChunk(cdata);
144                let buf = postcard::to_allocvec(&req)?;
145                stream_upload_chunk.send_u64_le(buf.len() as u64).await?;
146                stream_upload_chunk.send(&buf).await?;
147            }
148            tx4.send((chunk, has_chunk)).await;
149        }
150        Ok(())
151    };
152
153    let task5 = async {
154        stats.replace(State::PushingChunks {
155            cur: 0,
156            total,
157            tsize: 0,
158            tcsize: 0,
159        });
160
161        let mut chunks = Vec::new();
162        while let Some((chunk, has_chunk)) = rx4.recv().await {
163            if !has_chunk {
164                let len = stream_upload_chunk.recv_u64_le().await?;
165                assert_eq!(len, 0);
166            }
167            chunks.push(chunk);
168
169            total_size += chunk.size;
170            total_csize += chunk.csize;
171
172            stats.replace(State::PushingChunks {
173                cur: chunks.len(),
174                total,
175                tsize: total_size,
176                tcsize: total_csize,
177            });
178            udp.send(
179                server_address,
180                &postcard::to_allocvec(&UdpRequest::ActionProgress(chunks.len(), total))?,
181            )
182            .await?;
183        }
184        Ok(chunks)
185    };
186
187    let ((), (), (), (), chunk_hashes) = futures::try_join!(task1, task2, task3, task4, task5)?;
188
189    save_image(
190        &stream_upload_chunk,
191        Image {
192            boot_option_id: boid,
193            boot_entry: bo_command,
194            disk: chunk_hashes,
195        },
196    )
197    .await?;
198
199    stream_get_csize.close_send().await;
200    stream_get_csize.force_close().await;
201
202    stream_upload_chunk.close_send().await;
203    // TODO(virv): this could be better
204    stream_upload_chunk.force_close().await;
205
206    log::info!("image saved. Total size {total_size}, total csize {total_csize}");
207
208    Ok(())
209}
