10fn le16(buf: &[u8], lo: usize) -> u16 {
11    (0..2).map(|i| (buf[lo + i] as u16) << (8 * i)).sum()
12}
13
14fn le32(buf: &[u8], lo: usize) -> u32 {
15    (0..4).map(|i| (buf[lo + i] as u32) << (8 * i)).sum()
16}
17
18fn le64(buf: &[u8], lo: usize) -> u64 {
19    (0..8).map(|i| (buf[lo + i] as u64) << (8 * i)).sum()
20}
21
22fn le64_32_32(buf: &[u8], lo: usize, hi: usize) -> u64 {
23    (0..4)
24        .map(|i| ((buf[lo + i] as u64) << (8 * i)) + ((buf[hi + i] as u64) << (8 * i + 32)))
25        .sum()
26}

28async fn get_ext4_chunks(disk: &Disk, start: u64, end: u64) -> Result<Option<Vec<ChunkInfo>>> {
29    fn has_superblock(group: usize) -> bool {
30        if group <= 1 {
31            return true;
32        }
33
34        for d in [3, 5, 7] {
35            let mut p = 1;
36            while p < group {
37                p *= d;
38            }
39            if p == group {
40                return true;
41            }
42        }
43
44        false
45    }
46
47    if start + 2048 > end {
48        // Not an ext4 partition.
49        return Ok(None);
50    }
51    // Read superblock.
52    let mut superblock = [0; 1024];
53    disk.read(start + 1024, &mut superblock).await?;
54
55    let magic = le16(&superblock, 0x38);
56    if magic != 0xEF53 {
57        return Ok(None);
58    }
59
60    let feature_incompat = le32(&superblock, 0x60);
61    if feature_incompat & 0x80 == 0 {
62        // INCOMPAT_64BIT flag
63        return Ok(None);
64    }
65
66    let feature_ro_compat = le32(&superblock, 0x64);
67    if feature_ro_compat & 0x1 == 0 {
68        // RO_COMPAT_SPARSE_SUPER flag
69        return Ok(None);
70    }
71
72    let blocks_count = le64_32_32(&superblock, 0x4, 0x150);
73    let log_block_size = le32(&superblock, 0x18);
74    assert!(blocks_count.checked_shl(10 + log_block_size).is_some());
75    let block_size = 1u64 << (10 + log_block_size);
76
77    let blocks_per_group = le32(&superblock, 0x20) as u64;
78    let groups = blocks_count.div_ceil(blocks_per_group);
79
80    let first_data_block = le32(&superblock, 0x14) as u64;
81    let desc_size = le16(&superblock, 0xfe) as u64;
82    let reserved_gdt_blocks = le16(&superblock, 0xce);
83
84    let blocks_for_special_group =
85        1 + (desc_size * groups).div_ceil(block_size) as usize + reserved_gdt_blocks as usize;
86
87    let mut group_descriptors = vec![0; (desc_size * groups) as usize];
88    let mut bitmap = vec![0; block_size as usize];
89    disk.read(
90        start + block_size * (first_data_block + 1),
91        &mut group_descriptors,
92    )
93    .await?;
94
95    let mut ans = Vec::new();
96
97    for (group, group_descriptor) in group_descriptors.chunks(desc_size as usize).enumerate() {
98        let flags = le16(group_descriptor, 0x12);
99        if flags & 0x2 != 0 {
100            // EXT4_BG_BLOCK_UNINIT
101            if has_superblock(group) {
102                for block in 0..blocks_for_special_group {
103                    if group * blocks_per_group as usize + block < blocks_count as usize {
104                        ans.push(ChunkInfo {
105                            start: block_size as usize
106                                * (group * blocks_per_group as usize + block),
107                            size: block_size as usize,
108                        });
109                    }
110                }
111            }
112        } else {
113            let block_bitmap = le64_32_32(group_descriptor, 0x0, 0x20);
114
115            disk.read(start + block_size * block_bitmap, &mut bitmap)
116                .await?;
117
118            for block in 0..8 * block_size as usize {
119                let is_used = bitmap[block / 8] >> (block % 8) & 1 != 0;
120                if is_used && group * blocks_per_group as usize + block < blocks_count as usize {
121                    ans.push(ChunkInfo {
122                        start: block_size as usize * (group * blocks_per_group as usize + block),
123                        size: block_size as usize,
124                    });
125                }
126            }
127        }
128    }
129
130    Ok(Some(ans))
131}

133async fn get_ntfs_chunks(disk: &Disk, start: u64, end: u64) -> Result<Option<Vec<ChunkInfo>>> {
134    if end - start < 512 {
135        return Ok(None);
136    }
137
138    let mut boot_sector = [0u8; 512];
139    disk.read(start, &mut boot_sector).await?;
140
141    if &boot_sector[3..11] != b"NTFS    " {
142        return Ok(None);
143    }
144
145    let bytes_per_sector = le16(&boot_sector, 0x0b) as usize;
146
147    let sectors_per_cluster = match boot_sector[0x0d] {
148        x @ 0..=127 => x as usize,
149        x @ 225..=255 => 1 << -(x as i8),
150        x @ 128..=224 => panic!("too many sectors per cluster: {}", x),
151    };
152    let bytes_per_cluster = bytes_per_sector * sectors_per_cluster;
153    let num_clusters = (end as usize - start as usize).div_ceil(bytes_per_cluster);
154
155    let bytes_per_file_record = match boot_sector[0x40] {
156        x @ 0..=127 => x as usize * bytes_per_cluster,
157        x @ 225..=255 => 1 << -(x as i8),
158        x @ 128..=224 => panic!("too many bytes per file record: {}", x),
159    };
160
161    let mft_cluster_number = le64(&boot_sector, 0x30) as usize;
162    let mft_address = bytes_per_cluster * mft_cluster_number;
163
164    let bitmap_entry_address = mft_address + 6 * bytes_per_file_record;
165    let mut bitmap_entry = [0u8; 1024];
166    disk.read(start + bitmap_entry_address as u64, &mut bitmap_entry)
167        .await
168        .map_err(|e| Error::Generic(format!("failed to read bitmap entry: {}", e)))?;
169
170    let mut attribute_offset = le16(&bitmap_entry, 0x14) as usize;
171    while le32(&bitmap_entry, attribute_offset) != 0x80 {
172        attribute_offset += le32(&bitmap_entry, attribute_offset + 4) as usize;
173    }
174
175    let non_resident_flag = bitmap_entry[attribute_offset + 8];
176    assert_eq!(non_resident_flag, 1);
177
178    let mut start_vcn = le64(&bitmap_entry, attribute_offset + 0x10) as usize;
179    let last_vcn = le64(&bitmap_entry, attribute_offset + 0x18) as usize;
180    let mut data_run_offset =
181        attribute_offset + le16(&bitmap_entry, attribute_offset + 0x20) as usize;
182
183    let mut cnt = 0;
184    let mut chunks = Vec::new();
185
186    while start_vcn <= last_vcn {
187        let ctrl_byte = bitmap_entry[data_run_offset];
188
189        let length_len = (ctrl_byte & 0x0f) as usize;
190        let length =
191            (le64(&bitmap_entry, data_run_offset + 1) & ((1 << (8 * length_len)) - 1)) as usize;
192
193        let offset_len = (ctrl_byte >> 4) as usize;
194        let offset = (le64(&bitmap_entry, data_run_offset + 1 + length_len)
195            & ((1 << (8 * offset_len)) - 1)) as usize;
196
197        let mut buf = vec![0u8; bytes_per_cluster];
198        for i in 0..length {
199            let x = start + (offset + i) as u64 * bytes_per_cluster as u64;
200            disk.read(x, &mut buf).await.map_err(|e| {
201                Error::Generic(format!("failed to read bitmap content at {}: {}", x, e))
202            })?;
203
204            for &byte in &buf {
205                for bit in 0..8 {
206                    if cnt < num_clusters as u64 {
207                        if byte >> bit & 1 != 0 {
208                            chunks.push(ChunkInfo {
209                                start: cnt as usize * bytes_per_cluster,
210                                size: bytes_per_cluster,
211                            });
212                        }
213                        cnt += 1;
214                    }
215                }
216            }
217        }
218
219        start_vcn += length;
220        data_run_offset += 1 + length_len + offset_len;
221    }
222
223    Ok(Some(chunks))
224}

226async fn get_swap_chunks(disk: &Disk, start: u64, end: u64) -> Result<Option<Vec<ChunkInfo>>> {
227    if end - start < 4096 {
228        return Ok(None);
229    }
230
231    let mut first_chunk = [0u8; 4096];
232    disk.read(start, &mut first_chunk).await?;
233
234    if &first_chunk[4096 - 10..] != b"SWAPSPACE2" {
235        return Ok(None);
236    }
237
238    Ok(Some(vec![ChunkInfo {
239        start: 0,
240        size: 4096,
241    }]))
242}

244/// Returns chunks *relative to the start of the partition*.
245async fn parse_partition(disk: &Disk, start: u64, end: u64) -> Result<Vec<ChunkInfo>> {
246    if let Some(chunks) = get_ext4_chunks(disk, start, end).await? {
247        Ok(chunks)
248    } else if let Some(chunks) = get_ntfs_chunks(disk, start, end).await? {
249        Ok(chunks)
250    } else if let Some(chunks) = get_swap_chunks(disk, start, end).await? {
251        Ok(chunks)
252    } else {
253        Ok(vec![ChunkInfo {
254            start: 0,
255            size: (end - start) as usize,
256        }])
257    }
258}

260async fn parse_gpt(disk: &mut Disk) -> Result<Option<Vec<ChunkInfo>>> {
261    let disk_size = disk.size() as usize;
262    let Ok(partitions) = disk.partitions() else {
263        return Ok(None);
264    };
265
266    let mut pos = 0usize;
267    let mut chunks = vec![];
268    for partition in partitions {
269        let begin = partition.byte_start as usize;
270        let end = partition.byte_end as usize;
271
272        if pos < begin {
273            chunks.push(ChunkInfo {
274                start: pos,
275                size: (begin - pos),
276            });
277        }
278
279        let part_chunks = parse_partition(disk, begin as u64, end as u64).await?;
280        for ChunkInfo { start, size } in part_chunks {
281            chunks.push(ChunkInfo {
282                start: start + begin,
283                size,
284            });
285        }
286
287        pos = end;
288    }
289
290    if pos < disk_size {
291        chunks.push(ChunkInfo {
292            start: pos,
293            size: disk_size - pos,
294        });
295    }
296
297    Ok(Some(chunks))
298}
