1use crate::os::{
2    error::{Error, Result},
3    UefiOS, PACKET_SIZE,
4};
5use alloc::{boxed::Box, rc::Rc, vec::Vec};
6use core::{cell::RefCell, net::SocketAddrV4};
7use futures::future::{select, Either};
8use pixie_shared::{HintPacket, RegistrationInfo, TcpRequest, HINT_PORT};
9use uefi::proto::console::text::{Color, Key, ScanCode};
10
11#[derive(Debug, Default)]
12struct Data {
13    station: RegistrationInfo,
14    selected: usize,
15}
16
17pub async fn register(os: UefiOS, server_addr: SocketAddrV4) -> Result<()> {
18    let data = Rc::new(RefCell::new(Data::default()));
19    let data2 = data.clone();
20
21    os.set_ui_drawer(move |os| {
22        let data2 = data2.borrow();
23        os.write_with_color(
24            &format!("Group:  {}\n", data2.station.group),
25            if data2.selected == 0 {
26                Color::Yellow
27            } else {
28                Color::White
29            },
30            Color::Black,
31        );
32        os.write_with_color(
33            &format!("Row:    {}\n", data2.station.row),
34            if data2.selected == 1 {
35                Color::Yellow
36            } else {
37                Color::White
38            },
39            Color::Black,
40        );
41        os.write_with_color(
42            &format!("Column: {}\n", data2.station.col),
43            if data2.selected == 2 {
44                Color::Yellow
45            } else {
46                Color::White
47            },
48            Color::Black,
49        );
50        os.write_with_color(
51            &format!("Image:  {}\n", data2.station.image),
52            if data2.selected == 3 {
53                Color::Yellow
54            } else {
55                Color::White
56            },
57            Color::Black,
58        );
59    });
60
61    let udp = os.udp_bind(Some(HINT_PORT)).await?;
62    let mut buf = [0; PACKET_SIZE];
63
64    let mut hint = true;
65    let mut images = Vec::new();
66    let mut groups = Vec::new();
67
68    loop {
69        let key = if hint {
70            loop {
71                let recv = Box::pin(udp.recv(&mut buf));
72                let key = Box::pin(os.read_key());
73                match select(recv, key).await {
74                    Either::Left(((buf, _), _)) => {
75                        let hint: HintPacket = postcard::from_bytes(buf)?;
76                        data.borrow_mut().station = hint.station;
77                        images = hint.images;
78                        groups = hint.groups.into_iter().map(|(k, _)| k).collect();
79                        os.force_ui_redraw();
80                    }
81                    Either::Right((key, _)) => {
82                        hint = false;
83                        break key?;
84                    }
85                }
86            }
87        } else {
88            os.read_key().await?
89        };
90
91        if key == Key::Special(ScanCode::DOWN) {
92            let mut data = data.borrow_mut();
93            data.selected = (data.selected + 1).min(3);
94        }
95        if key == Key::Special(ScanCode::UP) {
96            let mut data = data.borrow_mut();
97            data.selected = data.selected.saturating_sub(1);
98        }
99        if key == Key::Special(ScanCode::LEFT) {
100            let mut data = data.borrow_mut();
101            match data.selected {
102                0 => {
103                    data.station.group = groups
104                        .iter()
105                        .rev()
106                        .cycle()
107                        .skip_while(|g| **g != data.station.group)
108                        .nth(1)
109                        .unwrap()
110                        .clone()
111                }
112                1 => data.station.row = data.station.row.wrapping_sub(1),
113                2 => data.station.col = data.station.col.wrapping_sub(1),
114                3 => {
115                    data.station.image = images[(images
116                        .iter()
117                        .position(|x| x == &data.station.image)
118                        .ok_or(Error::Generic("Invalid image name".into()))?
119                        + images.len()
120                        - 1)
121                        % images.len()]
122                    .clone();
123                }
124                _ => unreachable!(),
125            }
126        }
127        if key == Key::Special(ScanCode::RIGHT) {
128            let mut data = data.borrow_mut();
129            match data.selected {
130                0 => {
131                    data.station.group = groups
132                        .iter()
133                        .cycle()
134                        .skip_while(|g| **g != data.station.group)
135                        .nth(1)
136                        .unwrap()
137                        .clone()
138                }
139                1 => data.station.row = data.station.row.wrapping_add(1),
140                2 => data.station.col = data.station.col.wrapping_add(1),
141                3 => {
142                    data.station.image = images[(images
143                        .iter()
144                        .position(|x| x == &data.station.image)
145                        .ok_or(Error::Generic("Invalid image name".into()))?
146                        + 1)
147                        % images.len()]
148                    .clone();
149                }
150                _ => unreachable!(),
151            }
152        }
153        if key == Key::Printable('\r'.try_into().unwrap()) {
154            break;
155        }
156        os.force_ui_redraw();
157    }
158
159    let msg = TcpRequest::Register(data.borrow().station.clone());
160    let buf = postcard::to_allocvec(&msg)?;
161    let stream = os.connect(server_addr).await?;
162    stream.send_u64_le(buf.len() as u64).await?;
163    stream.send(&buf).await?;
164    let len = stream.recv_u64_le().await?;
165    assert_eq!(len, 0);
166    stream.close_send().await;
167    // TODO(virv): this could be better
168    stream.force_close().await;
169
170    log::info!("Registration successful! {:?}", data.borrow().station);
171
172    Ok(())
173}
