19fn open_disk(handle: Handle) -> Result<(ScopedProtocol<DiskIo>, ScopedProtocol<BlockIO>)> {
20    let image_handle = uefi::boot::image_handle();
21    let bio = unsafe {
22        uefi::boot::open_protocol::<BlockIO>(
23            OpenProtocolParams {
24                agent: image_handle,
25                controller: None,
26                handle,
27            },
28            uefi::boot::OpenProtocolAttributes::GetProtocol,
29        )?
30    };
31    let proto = uefi::boot::open_protocol_exclusive::<DiskIo>(handle)?;
32    Ok((proto, bio))
33}

49// TODO(veluca): consider making parts of this actually async, i.e. by using DiskIo2/BlockIO2 if
50// available; support having more than one disk.
51impl Disk {
52    pub fn new(os: UefiOS) -> Disk {
53        let handle = uefi::boot::find_handles::<DiskIo>()
54            .unwrap()
55            .into_iter()
56            .find(|handle| {
57                let op = open_disk(*handle);
58                if let Ok((_, block)) = op {
59                    if block.media().is_media_present() {
60                        return true;
61                    }
62                }
63                false
64            })
65            .expect("Disk not found");
66
67        let (disk, block) = open_disk(handle).unwrap();
68        Disk { disk, block, os }
69    }
70
71    pub fn size(&self) -> u64 {
72        self.block.media().block_size() as u64 * (self.block.media().last_block() + 1)
73    }
74
75    pub async fn flush(&mut self) {
76        self.block.flush_blocks().unwrap();
77    }
78
79    pub async fn read(&self, offset: u64, buf: &mut [u8]) -> Result<()> {
80        self.os.schedule().await;
81        Ok(self
82            .disk
83            .read_disk(self.block.media().media_id(), offset, buf)?)
84    }
85
86    pub async fn write(&mut self, offset: u64, buf: &[u8]) -> Result<()> {
87        self.os.schedule().await;
88        Ok(self
89            .disk
90            .write_disk(self.block.media().media_id(), offset, buf)?)
91    }
92
93    pub fn partitions(&mut self) -> Result<Vec<DiskPartition>> {
94        let block_size = self.block_size().to_u64();
95        let mut disk = gpt_disk_io::Disk::new(self).map_err(|e| Error::Generic(e.to_string()))?;
96        let mut buf = [0; 1 << 14];
97        let header = disk
98            .read_primary_gpt_header(&mut buf)
99            .map_err(|e| Error::Generic(e.to_string()))?;
100        // TODO(veluca): bubble up this error.
101        let part_array_layout = header
102            .get_partition_entry_array_layout()
103            .map_err(|e| Error::Generic(e.to_string()))?;
104        let mut buf = [0; 1 << 14];
105        let x = disk
106            .gpt_partition_entry_array_iter(part_array_layout, &mut buf)
107            .map_err(|e| Error::Generic(e.to_string()))?
108            .filter_map(|part| {
109                let part = if let Err(err) = part {
110                    return Some(Err(err));
111                } else {
112                    part.unwrap()
113                };
114                if part.is_used() {
115                    let part_guid = part.unique_partition_guid;
116                    Some(Ok(DiskPartition {
117                        byte_start: part.starting_lba.to_u64() * block_size,
118                        byte_end: (part.ending_lba.to_u64() + 1) * block_size,
119                        guid: part_guid.to_string(),
120                        name: part.name.to_string(),
121                    }))
122                } else {
123                    None
124                }
125            })
126            .collect::<Result<_, _>>()
127            .map_err(|e| Error::Generic(e.to_string()))?;
128
129        Ok(x)
130    }
131}

133impl gpt_disk_io::BlockIo for &mut Disk {

136    fn block_size(&self) -> BlockSize {

140        Ok(self.block.media().last_block() + 1)

143        self.disk.read_disk(
144            self.block.media().media_id(),
145            self.block.media().block_size() as u64 * start_lba.0,
146            dst,
147        )?;
148        Ok(())
149    }
150    fn write_blocks(&mut self, _start_lba: Lba, _src: &[u8]) -> Result<()> {
151        unreachable!();
152    }

154        Ok(self.block.flush_blocks()?)
