57struct BytesFmt(u64);

59impl Display for BytesFmt {
60    fn fmt(&self, f: &mut fmt::Formatter<'_>) -> fmt::Result {
61        if self.0 < (1 << 10) {
62            write!(f, "{}B", self.0)
63        } else if self.0 < (1 << 20) {
64            write!(f, "{:.2}KiB", self.0 as f64 / 1024.0)
65        } else if self.0 < (1 << 30) {
66            write!(f, "{:.2}MiB", self.0 as f64 / (1 << 20) as f64)
67        } else {
68            write!(f, "{:.2}GiB", self.0 as f64 / (1 << 30) as f64)
69        }
70    }
71}

93    pub fn write_with_color(&mut self, msg: &str, fg: Color, bg: Color) {
94        let lines: Vec<_> = msg.split('\n').collect();
95        for (n, line) in lines.iter().enumerate() {
96            self.ui_buf.push((line.to_string(), fg, bg));
97            self.ui_pos += line.len();
98            if n != lines.len() - 1 {
99                let cols = self.cols();
100                let colp = self.ui_pos % cols;
101                let n = cols - colp;
102                self.ui_buf
103                    .push((String::from_utf8(vec![0x20; n]).unwrap(), fg, bg));
104                self.ui_pos += n;
105            }
106        }
107    }
108
109    pub fn maybe_advance_to_col(&mut self, col: usize) {
110        let (fg, bg) = if let Some((_, f, b)) = self.ui_buf[..].last() {
111            (*f, *b)
112        } else {
113            (Color::White, Color::Black)
114        };
115        let cols = self.cols();
116        let colp = self.ui_pos % cols;
117        let n = col - colp;
118        if colp < col {
119            self.ui_buf
120                .push((String::from_utf8(vec![0x20; n]).unwrap(), fg, bg));
121            self.ui_pos += n;
122        }
123    }
124
125    pub fn flush_ui_buf(&mut self) {
126        self.vga.set_cursor_position(0, 0).unwrap();
127        let mode = self.vga.current_mode().unwrap().unwrap();
128        let (cols, rows) = (mode.columns(), mode.rows());
129        for (msg, fg, bg) in self.ui_buf.drain(..) {
130            self.vga.set_color(fg, bg).unwrap();
131            write!(self.vga, "{msg}").unwrap();
132        }
133        self.vga.set_color(Color::White, Color::Black).unwrap();
134        if self.ui_pos + 1 < cols * rows {
135            // Clear any remaining chars.
136            let n = cols * rows - self.ui_pos - 1;
137            write!(self.vga, "{}", String::from_utf8(vec![0x20; n]).unwrap()).unwrap();
138        }
139        self.ui_pos = 0;
140    }

157    pub fn start<F, Fut>(mut f: F) -> !
158    where
159        F: FnMut(UefiOS) -> Fut + 'static,
160        Fut: Future<Output = Result<()>>,
161    {
162        // Never call this function twice.
163        assert!(OS.borrow().is_none());
164
165        uefi::helpers::init().unwrap();
166
167        // Ensure we never exit boot services.
168        // SAFETY: the callback panics on exit from boot services, and thus handles exit from boot
169        // services correctly by definition.
170        unsafe {
171            uefi::boot::create_event(
172                EventType::SIGNAL_EXIT_BOOT_SERVICES,
173                Tpl::NOTIFY,
174                Some(exit_boot_services),
175                None,
176            )
177            .unwrap();
178        }
179
180        let timer = Timer::new();
181        let rng = Rng::new();
182
183        let input_handles = uefi::boot::find_handles::<Input>().unwrap();
184        let input = uefi::boot::open_protocol_exclusive::<Input>(input_handles[0]).unwrap();
185
186        let serial = uefi::boot::find_handles::<Serial>()
187            .ok()
188            .map(|handles| uefi::boot::open_protocol_exclusive::<Serial>(handles[0]).unwrap());
189
190        let vga_handles = uefi::boot::find_handles::<Output>().unwrap();
191        let mut vga = uefi::boot::open_protocol_exclusive::<Output>(vga_handles[0]).unwrap();
192
193        vga.clear().unwrap();
194
195        *OS.borrow_mut() = Some(UefiOSImpl {
196            timer,
197            rng,
198            tasks: Vec::new(),
199            input,
200            vga,
201            serial,
202            net: None,
203            messages: VecDeque::new(),
204            ui_buf: vec![],
205            ui_pos: 0,
206            ui_drawer: None,
207        });
208
209        let os = UefiOS { cant_build: () };
210
211        log::set_logger(&UefiOS { cant_build: () }).unwrap();
212        log::set_max_level(log::LevelFilter::Trace);
213
214        let net = NetworkInterface::new(os);
215        os.borrow_mut().net = Some(net);
216
217        os.spawn("init", async move {
218            loop {
219                if let Err(err) = f(os).await {
220                    log::error!("Error: {err:?}");
221                }
222            }
223        });
224
225        os.spawn("[watchdog]", async move {
226            loop {
227                let err = uefi::boot::set_watchdog_timer(300, 0x10000, None);
228
229                if let Err(err) = err {
230                    if err.status() != Status::UNSUPPORTED {
231                        log::error!("Error disabling watchdog: {err:?}");
232                    }
233
234                    break;
235                }
236
237                os.sleep_us(30_000_000).await;
238            }
239        });
240
241        os.spawn(
242            "[net_poll]",
243            poll_fn(move |cx| {
244                let (mut net, timer) =
245                    RefMut::map_split(os.borrow_mut(), |os| (&mut os.net, &mut os.timer));
246                net.as_mut().unwrap().poll(&timer);
247                // TODO(veluca): figure out whether we can suspend the task.
248                cx.waker().wake_by_ref();
249                Poll::Pending
250            }),
251        );
252
253        os.spawn("[net_speed]", async move {
254            let mut prx = 0;
255            let mut ptx = 0;
256            let mut ptm = os.timer().instant();
257            loop {
258                {
259                    let now = os.timer().instant();
260                    let dt = (now - ptm).total_micros() as f64 / 1_000_000.0;
261                    ptm = now;
262
263                    let mut net = os.net();
264                    net.vrx = ((net.rx - prx) as f64 / dt) as u64;
265                    prx = net.rx;
266                    net.vtx = ((net.tx - ptx) as f64 / dt) as u64;
267                    ptx = net.tx;
268                }
269                os.sleep_us(1_000_000).await;
270            }
271        });
272
273        os.spawn("[draw_ui]", async move {
274            loop {
275                os.draw_ui();
276                os.sleep_us(1_000_000).await;
277            }
278        });
279
280        Executor::run(os)
281    }

333    pub fn sleep_us(self, us: u64) -> PollFn<impl FnMut(&mut Context<'_>) -> Poll<()>> {
334        let tgt = self.timer().micros() as u64 + us;
335        poll_fn(move |cx| {
336            let now = self.timer().micros() as u64;
337            if now >= tgt {
338                Poll::Ready(())
339            } else {
340                // TODO(veluca): actually suspend the task.
341                cx.waker().wake_by_ref();
342                Poll::Pending
343            }
344        })

359        vendor: &VariableVendor,
360    ) -> Result<(Vec<u8>, VariableAttributes)> {
361        // name.len() should be enough, but...

398    /// Find the topmost device that implements this protocol.
399    fn handle_on_device<P: Protocol>(&self, device: &DevicePath) -> Option<Handle> {
400        for i in 0..device.node_iter().count() {
401            let mut buf = vec![];
402            let mut dev = DevicePathBuilder::with_vec(&mut buf);
403            for node in device.node_iter().take(i + 1) {
404                dev = dev.push(&node).unwrap();
405            }
406            let mut dev = dev.finalize().unwrap();
407            if let Ok(h) = uefi::boot::locate_device_path::<P>(&mut dev) {
408                return Some(h);
409            }
410        }
411        None
412    }

446    fn draw_ui(&self) {
447        // Write the header.
448        {
449            let time = self.timer().micros() as f32 * 0.000_001;
450            let ip = self.net().ip();
451            let mut os = self.borrow_mut();
452
453            let mode = os.vga.current_mode().unwrap().unwrap();
454            let cols = mode.columns();
455
456            os.write_with_color(&format!("uptime: {time:10.1}s"), Color::White, Color::Black);
457            os.maybe_advance_to_col(cols / 3);
458
459            if let Some(ip) = ip {
460                os.write_with_color(&format!("IP: {ip}"), Color::White, Color::Black);
461            } else {
462                os.write_with_color("DHCP...", Color::Yellow, Color::Black);
463            }
464
465            os.maybe_advance_to_col(3 * cols / 5);
466
467            let vrx = os.net.as_ref().unwrap().vrx;
468            let vtx = os.net.as_ref().unwrap().vtx;
469            os.write_with_color(
470                &format!("rx: {}/s tx: {}/s\n\n", BytesFmt(vrx), BytesFmt(vtx)),
471                Color::White,
472                Color::Black,
473            );
474
475            os.tasks.sort_by_key(|t| -t.micros());
476            let tasks: Vec<_> = os.tasks.iter().take(7).cloned().collect();
477            for task in tasks {
478                os.write_with_color(task.name, Color::White, Color::Black);
479                os.maybe_advance_to_col(cols / 4);
480                os.write_with_color(
481                    &format!("{:7.3}s\n", task.micros() as f64 * 0.000_001),
482                    Color::White,
483                    Color::Black,
484                );
485            }
486
487            os.maybe_advance_to_col(cols);
488
489            // TODO(veluca): find a better solution.
490            let messages: Vec<_> = os.messages.iter().cloned().collect();
491
492            for (time, level, target, msg) in messages {
493                let fg_color = match level {
494                    log::Level::Trace => Color::Cyan,
495                    log::Level::Debug => Color::Blue,
496                    log::Level::Info => Color::Green,
497                    log::Level::Warn => Color::Yellow,
498                    log::Level::Error => Color::Red,
499                };
500                os.write_with_color(&format!("[{time:.1}s "), Color::White, Color::Black);
501                os.write_with_color(&format!("{level:5}"), fg_color, Color::Black);
502                os.write_with_color(&format!(" {target}] {msg}\n"), Color::White, Color::Black);
503            }
504            os.write_with_color("\n", Color::Black, Color::Black);
505        }
506        {
507            let ui = self.borrow_mut().ui_drawer.take();
508            if let Some(ui) = &ui {
509                ui(*self);
510            }
511            self.borrow_mut().ui_drawer = ui;
512        }
513        // Actually draw the changes.
514        self.borrow_mut().flush_ui_buf();
515    }

529    fn append_message(&self, time: f64, level: log::Level, target: &str, msg: String) {
530        {
531            let mut os = self.borrow_mut();
532
533            if let Some(serial) = &mut os.serial {
534                let style = match level {
535                    log::Level::Trace => anstyle::AnsiColor::Cyan.on_default(),
536                    log::Level::Debug => anstyle::AnsiColor::Blue.on_default(),
537                    log::Level::Info => anstyle::AnsiColor::Green.on_default(),
538                    log::Level::Warn => anstyle::AnsiColor::Yellow.on_default(),
539                    log::Level::Error => anstyle::AnsiColor::Red.on_default().bold(),
540                };
541                write!(
542                    serial,
543                    "[{time:.1}s {style}{level:5}{style:#} {target}] {msg}\r\n"
544                )
545                .unwrap();
546            }
547
548            os.messages.push_back((time, level, target.into(), msg));
549            const MAX_MESSAGES: usize = 5;
550            if os.messages.len() > MAX_MESSAGES {
551                os.messages.pop_front();
552            }
553        }
554        self.force_ui_redraw();
555    }
