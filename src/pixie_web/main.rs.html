13fn send_req(url: String) {
14    let url = if url.starts_with("http") {
15        Url::parse(&url).expect("invalid url")
16    } else {
17        let location = window().location();
18        Url::parse(&location.href().expect("no href"))
19            .expect("invalid href")
20            .join(&url)
21            .expect("invalid url")
22    };
23    spawn_local(async move {
24        reqwest::get(url.clone())
25            .await
26            .unwrap_or_else(|_| panic!("Request to {} failed", url));
27    });
28}

30struct Bytes(u64);
31
32impl fmt::Display for Bytes {
33    fn fmt(&self, f: &mut fmt::Formatter<'_>) -> fmt::Result {
34        if self.0 < 1024 {
35            write!(f, "{} B", self.0)
36        } else if self.0 < 1024 * 1024 {
37            write!(f, "{:.2} KiB", self.0 as f64 / 1024.0)
38        } else if self.0 < 1024 * 1024 * 1024 {
39            write!(f, "{:.2} MiB", self.0 as f64 / 1024.0 / 1024.0)
40        } else {
41            write!(f, "{:.2} GiB", self.0 as f64 / 1024.0 / 1024.0 / 1024.0)
42        }
43    }
44}

298#[component]
299fn App() -> impl IntoView {
300    let (config, set_config) = create_signal(None::<Config>);
301    let (hostmap, set_hostname) = create_signal(None::<HashMap<Ipv4Addr, String>>);
302    let (units, set_units) = create_signal(None::<Vec<Unit>>);
303    let (image_stats, set_image_stats) = create_signal(None::<ImageStat>);
304
305    let images = Signal::derive(move || {
306        config
307            .get()
308            .map(|x| x.images.clone())
309            .unwrap_or_else(Vec::new)
310    });
311
312    let location =
313        Url::parse(&window().location().href().expect("no href")).expect("invalid url href");
314    let status_url = location
315        .join("admin/status")
316        .expect("could not make relative URL");
317
318    let handle_message = move |msg| match msg {
319        StatusUpdate::Units(mut u) => {
320            u.sort_by_key(|x| x.static_ip());
321            set_units.set(Some(u));
322        }
323        StatusUpdate::Config(c) => set_config.set(Some(c)),
324        StatusUpdate::HostMap(h) => set_hostname.set(Some(h)),
325        StatusUpdate::ImageStats(i) => set_image_stats.set(Some(i)),
326    };
327
328    spawn_local(async move {
329        let stream = reqwest::get(status_url)
330            .await
331            .expect("could not connect to server");
332        let mut buf = vec![];
333        stream
334            .bytes_stream()
335            .for_each(|x| {
336                let data = x.unwrap();
337                let mut data = &data[..];
338                while let Some((newline_pos, _)) =
339                    data.iter().enumerate().find(|(_, x)| **x == b'\n')
340                {
341                    buf.extend_from_slice(&data[..newline_pos]);
342                    let msg: StatusUpdate =
343                        serde_json::from_slice(&buf).expect("invalid message from server");
344                    buf.clear();
345                    handle_message(msg);
346                    data = &data[newline_pos + 1..];
347                }
348                buf.extend_from_slice(data);
349
350                async {}
351            })
352            .await;
353    });
354
355    let time = use_timestamp();
356    let time_in_seconds = create_memo(move |_| (time.get() * 0.001) as i64);
357
358    let render_group = move |id: u8| {
359        let group_name = create_memo(move |_| {
360            config
361                .get()
362                .unwrap()
363                .groups
364                .get_by_second(&id)
365                .unwrap()
366                .clone()
367        });
368        let units = create_memo(move |_| -> Vec<_> {
369            units
370                .get()
371                .unwrap_or_else(Vec::new)
372                .iter()
373                .filter(|x| x.group == id)
374                .cloned()
375                .collect()
376        });
377        let hostmap = Signal::derive(move || hostmap.get().unwrap_or_else(HashMap::new));
378        view! { <Group units group_name images hostmap time=time_in_seconds/> }.into_view()
379    };
380
381    view! {
382        <Images images=image_stats/>
383        <For
384            each=move || {
385                config.get().clone().into_iter().flat_map(|x| x.groups.into_iter().map(|x| x.1))
386            }
387            key=|x| *x
388            children=render_group
389        />
390    }
391}
