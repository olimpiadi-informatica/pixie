49/// An image is given by the list of chunks of the disk, the index of the boot entry that boots it,
50/// and the contents of that boot entry.
51#[derive(Debug, Clone, Serialize, Deserialize)]
52pub struct Image {
53    pub boot_option_id: u16,
54    pub boot_entry: Vec<u8>,
55    pub disk: Vec<Chunk>,
56}
57
58impl Image {
59    /// Computes the size in bytes of an image.
60    ///
61    /// Repeated chunks are counted many times.
62    pub fn size(&self) -> u64 {
63        self.disk.iter().map(|chunk| chunk.size as u64).sum()
64    }
65
66    /// Computes the compressed size in bytes of an image.
67    ///
68    /// Repeated chunks are counted once.
69    pub fn csize(&self) -> u64 {
70        let mut chunks: Vec<_> = self
71            .disk
72            .iter()
73            .map(|chunk| (chunk.hash, chunk.csize))
74            .collect();
75        chunks.sort_unstable_by_key(|(hash, _)| *hash);
76        chunks.dedup_by_key(|(hash, _)| *hash);
77        chunks.into_iter().map(|(_, size)| size as u64).sum()
78    }
79}

117#[derive(Debug, Clone, Copy, Serialize, Deserialize, PartialEq, Eq)]
118#[serde(rename_all = "lowercase")]
119pub enum Action {
120    /// Reboot into the OS.
121    Reboot,
122    /// Shutdown the machine.
123    Shutdown,
124    /// Register the machine.
125    Register,
126    /// Make an image out of the current content of the disk and store it in the server database.
127    Store,
128    /// Fetch the image from the server and flash it to the disk.
129    Flash,
130    /// Wait for another command.
131    Wait,
132}
133
134impl Display for Action {
135    fn fmt(&self, fmt: &mut core::fmt::Formatter<'_>) -> Result<(), core::fmt::Error> {
136        match self {
137            Action::Reboot => write!(fmt, "reboot"),
138            Action::Shutdown => write!(fmt, "shutdown"),
139            Action::Register => write!(fmt, "register"),
140            Action::Store => write!(fmt, "store"),
141            Action::Flash => write!(fmt, "flash"),
142            Action::Wait => write!(fmt, "wait"),
143        }
144    }
145}

147/// A request for the udp server.
148///
149/// To send a request to the server it must be serialized with postcard and sent
150/// in a single udp packet at the address `<SERVER_IP>:25640`.
151#[derive(Debug, Clone, Serialize, Deserialize)]
152pub enum UdpRequest {
153    /// Used for server discovery.
154    /// To be sent in broadcast over the lan.
155    /// The server will reply with an empty packet.
156    Discover,
157    /// Sets the progress of the client for the current action.
158    ActionProgress(usize, usize),
159    /// Requests the given chunks to be broadcasted by the server.
160    RequestChunks(Vec<ChunkHash>),
161}

163/// A request for the tcp server.
164///
165/// Over a single tcp connection multiple request can be sent.
166/// Each request is encoded as:
167/// - request length with 8 bytes;
168/// - request content encoded with postcard;
169///
170/// To each request the server will reply with a message in the same format:
171/// - request length with 8 bytes;
172/// - response content encoded with postcard;
173#[derive(Debug, Clone, Serialize, Deserialize)]
174pub enum TcpRequest {
175    /// Checks if the server contains the chunk in the database.
176    /// The server will reply with a bool.
177    HasChunk(ChunkHash),
178    /// Asks the server the [`Image`], the image name is deduced by the client configuration.
179    /// The server replies with the requested [`Image`].
180    GetImage,
181    /// Registers the client with the given info.
182    /// The response is empty.
183    Register(RegistrationInfo),
184    /// Uploads the given chunk to the server, the content is already compressed.
185    /// The response is empty.
186    UploadChunk(Vec<u8>),
187    /// Uploads the [`Image`] to the server, the image name is deduced by the client configuration.
188    /// The response is empty.
189    UploadImage(Image),
190    /// Asks the server for the client action to run.
191    /// The server will reply with an [`Action`].
192    GetAction,
193    /// Tells the server that the action is complete and can proced to the next action.
194    /// The response is emtpy.
195    ActionComplete,
196}
